		dir.mkdir();
		
		ITool from = new DeleteTool (new String[]{"testdirectory"});
		ITool to = new LsTool(new String[]{""});
		Vector<ITool> tools = new Vector<ITool>();
		tools.add(from);
		tools.add(to);
		
		pipingTool = new PipingTool(tools);
		pipingTool.pipe(from, to);
