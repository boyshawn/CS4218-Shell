	// Helper Functions
	public static void deleteFolder(File folder) throws IOException {
		File[] files = folder.listFiles();
		if (files != null) { // some JVMs return null for empty dirs
			for (File f : files) {
				if (f.isDirectory()) {
					deleteFolder(f);
				} else {
					f.delete();
				}
			}
			
			Files.delete(folder.toPath());
		} 
	}
